    # プロンプト選択
    import random
    prompt\_template = random.choice(PROMPT\_TEMPLATES[category])
    prompt = prompt\_template.format(color=color\_hex)
    
    try:
        # DALL-E 3で画像生成
        response = openai.Image.create(
            model="dall-e-3",
            prompt=prompt,
            size="1024x1024",
            quality="standard",  # "hd"に変更で高品質
            n=1
        )
        
        # 画像URL取得
        image\_url = response.data[0].url
        revised\_prompt = response.data[0].revised\_prompt
        
        # ファイル名生成
        timestamp = datetime.now().strftime("%Y%m%d\_%H%M%S")
        filename = f"{category}\_{color\_key}\_{timestamp}.png"
        filepath = Path(output\_dir) / filename
        
        # 画像ダウンロード
        import requests
        img\_response = requests.get(image\_url)
        with open(filepath, "wb") as f:
            f.write(img\_response.content)
        
        # メタデータ保存
        metadata = {
            "filename": filename,
            "category": category,
            "color": color\_hex,
            "original\_prompt": prompt,
            "revised\_prompt": revised\_prompt,
            "generated\_at": datetime.now().isoformat(),
            "url": image\_url
        }
        
        print(f"✓ 生成完了: {filename}")
        return metadata
        
    except openai.error.RateLimitError:
        print("⚠ API制限に達しました。60秒待機します...")
        time.sleep(60)
        return generate\_instagram\_image(category, color\_key, output\_dir)
    
    except Exception as e:
        print(f"✗ エラー: {e}")
        return None

def batch\_generate\_images(num\_images=10, categories=None):
    """画像を一括生成"""
    
    if categories is None:
        categories = list(PROMPT\_TEMPLATES.keys())
    
    generated\_list = []
    colors = list(BRAND\_COLORS.keys())
    
    print(f"Instagram画像生成開始（{num\_images}枚）...")
    
    for i in range(num\_images):
        # カテゴリとカラーをランダム選択
        import random
        category = random.choice(categories)
        color = random.choice(colors)
        
        metadata = generate\_instagram\_image(category, color)
        
        if metadata:
            generated\_list.append(metadata)
        
        # API制限対策（1分間に5リクエストまで）
        if (i + 1) % 5 == 0 and i < num\_images - 1:
            print("API制限対策: 60秒待機...")
            time.sleep(60)
    
    # 生成履歴をJSON保存
    log\_file = f"generation\_log\_{datetime.now().strftime('%Y%m%d\_%H%M%S')}.json"
    with open(log\_file, "w", encoding="utf-8") as f:
        json.dump(generated\_list, f, ensure\_ascii=False, indent=2)
    
    print(f"\n✓ 完了: {len(generated\_list)}/{num\_images}枚生成")
    print(f"✓ ログ保存: {log\_file}")
    
    return generated\_list
